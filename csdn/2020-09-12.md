        sum=sum+m*(m-1)/2;
        ll last=n-m+1;
        ll time=last/k;
        if(last%k)
            time++;
        sum+=(m-1)*time;
        sum+=(time-1)*last+(time-1)*(time-2)*(-k)/2;
        sum+=last%k;

        sum=n*(n+1)/2;
